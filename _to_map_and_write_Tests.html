  timeout = 500

  try_Http_Get = (next) =>
    @.GET (data)        => if data is null then (delay).invoke_After(next) else callback(data)
  run_Tests = (test_Count)=> if test_Count.empty() then @.GET (callback) else try_Http_Get ()->run_Tests(test_Count.splice(0,1))
